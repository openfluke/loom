# Bundles prebuilt Welvet C-ABI libraries from ../native/.
# Build loom/welvet/cabi first, then: bash tool/copy_native.sh

cmake_minimum_required(VERSION 3.10)
project(welvet_library VERSION 0.80.0 LANGUAGES C)

set(NATIVE_ROOT "${CMAKE_CURRENT_SOURCE_DIR}/../native")
set(WELVET_PREBUILT "")

if(ANDROID)
  if(ANDROID_ABI STREQUAL "arm64-v8a")
    set(WELVET_PREBUILT "${NATIVE_ROOT}/android/arm64-v8a/libwelvet.so")
  elseif(ANDROID_ABI STREQUAL "x86_64")
    set(WELVET_PREBUILT "${NATIVE_ROOT}/android/x86_64/libwelvet.so")
  else()
    message(FATAL_ERROR "Unsupported Android ABI: ${ANDROID_ABI}")
  endif()
elseif(WIN32)
  if(CMAKE_SIZEOF_VOID_P EQUAL 8 AND CMAKE_SYSTEM_PROCESSOR MATCHES "ARM64|aarch64")
    set(WELVET_PREBUILT "${NATIVE_ROOT}/windows_arm64/welvet.dll")
  else()
    set(WELVET_PREBUILT "${NATIVE_ROOT}/windows_amd64/welvet.dll")
  endif()
elseif(APPLE)
  if(IOS)
    # iOS: static Welvet.xcframework via ios/welvet.podspec (not CMake).
    set(WELVET_PREBUILT "${CMAKE_CURRENT_SOURCE_DIR}/welvet_ios_stub.c")
  else()
    if(EXISTS "${NATIVE_ROOT}/macos_universal/libwelvet.dylib")
      set(WELVET_PREBUILT "${NATIVE_ROOT}/macos_universal/libwelvet.dylib")
    elseif(EXISTS "${NATIVE_ROOT}/macos_arm64/libwelvet.dylib")
      set(WELVET_PREBUILT "${NATIVE_ROOT}/macos_arm64/libwelvet.dylib")
    else()
      set(WELVET_PREBUILT "${NATIVE_ROOT}/macos_amd64/libwelvet.dylib")
    endif()
  endif()
elseif(UNIX)
  if(CMAKE_SYSTEM_PROCESSOR MATCHES "aarch64|arm64")
    set(WELVET_PREBUILT "${NATIVE_ROOT}/linux_arm64/libwelvet.so")
  else()
    set(WELVET_PREBUILT "${NATIVE_ROOT}/linux_amd64/libwelvet.so")
  endif()
endif()

if(ANDROID OR WIN32 OR (APPLE AND NOT IOS) OR UNIX)
  if(WELVET_PREBUILT MATCHES "\\.c$")
    add_library(welvet SHARED "${WELVET_PREBUILT}")
    set_target_properties(welvet PROPERTIES OUTPUT_NAME "welvet")
  elseif(NOT EXISTS "${WELVET_PREBUILT}")
    message(FATAL_ERROR
      "Welvet prebuilt not found: ${WELVET_PREBUILT}\n"
      "Run: cd loom/welvet/dart && bash tool/copy_native.sh")
  else()
    message(STATUS "Welvet: using prebuilt ${WELVET_PREBUILT}")
    add_library(welvet SHARED IMPORTED GLOBAL)
    set_target_properties(welvet PROPERTIES IMPORTED_LOCATION "${WELVET_PREBUILT}")
  endif()
endif()
